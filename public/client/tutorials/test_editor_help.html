In the test editor you can use ChaiJs. Two bellow examples might be useful:

example 1:

Function in function editor:

function addTodo(todo){
throw new TypeError('IllegalArgumentException');
return todo;
};

Tests in test editor: 

var badFn = addTodo(null);
expect(badFn).to.throw(TypeError);

example 2:

Function in function editor:

function addTodo(todo){
var result = SaveObjectImplementation(todo);
return result;
};

Tests in test editor: 

 var param ={ "title": "coding","description": "work on the crowd cod",
"dueDate": "03/14/2018","dataStoreId": "todo3","userId": "emad.aghayi", "id": 1,"status": 1,
"groupId": "schoolworks","createdTime": "13:51","createdDate": "05/21/2018", "priority": 1,"address": "Fairfax,VA,US 22032"
};
var result1= addTodo(param);
var fetched = FetchObjectImplementation(1);
expect(fetched).to.deep.equal(result1);

TIP: While editing the test code, you can open the auto-complete by pressing OPTION+SPACE (Mac) or CTRL+SPACE (Win/Linux).
